std::vector<std::string> polishConverter(const std::string &s)
{
    std::map<char, char> precedence{
        {'#', 0},
        {'+', 1},
        {'-', 1},
        {'*', 2},
        {'/', 2},
        {'@', 2},
        {'(', 3},
    };
    std::vector<char> operatorStack;
    operatorStack.emplace_back('#');
    std::vector<std::string> expressionStack;
    std::string num;
    for (char c: s) {
        if (c >= '0' && c <= '9') {
            num += c;
            continue;
        }
        if (!num.empty()) {
            expressionStack.emplace_back(num);
            num = "";
        }
        if (c == '(') {
            operatorStack.emplace_back(c);
            continue;
        }
        if (c == ')') {
            while (true) {
                char back = operatorStack.back();
                operatorStack.pop_back();
                if (back == '(') {
                    break;
                }
                std::string tmp;
                tmp += back;
                expressionStack.emplace_back(tmp);
            }
            continue;
        }

        char back = operatorStack.back();
        while (back != '(' && precedence[back] > precedence[c]) {
            std::string tmp;
            tmp += back;
            expressionStack.emplace_back(tmp);
            operatorStack.pop_back();
            back = operatorStack.back();
        }
        operatorStack.emplace_back(c);
    }
    if (!num.empty()) {
        expressionStack.emplace_back(num);
        num = "";
    }
    while (!operatorStack.empty()) {
        char back = operatorStack.back();
        if (back == '#') {
            break;
        }
        std::string tmp;
        tmp += back;
        expressionStack.emplace_back(tmp);
        operatorStack.pop_back();
    }
    return expressionStack;
}

std::string polishCalculate(const std::vector<std::string> &expression)
{
    std::vector<std::string> numStack;
    for (const auto &item: expression) {
        if (item.size() > 1 || (item[0] >= '0' && item[0] <= '9')) {
            numStack.emplace_back(item);
            continue;
        }
        auto r = numStack.back();
        numStack.pop_back();
        auto l = numStack.back();
        numStack.pop_back();
        int ret;
        if (item == "+") {
            ret = add(l, r);
        } else if (item == "-") {
            ret = del(l, r);

        } else if (item == "*") {
            ret = mul(l, r);
        } else if (item == "/") {
            ret = dev(l, r);

        } else if (item == "@") {
            ret = dev(r, l);
        }
        numStack.emplace_back(std::to_string(ret));
    }
    return numStack.back();
}
